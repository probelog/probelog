public static void sort(List<Integer> numbers) {
    if (numbers.get(0) > numbers.get(1))
    for (int passLength = numbers.size(); passLength > 0; passLength--) pass(numbers, passLength);
        numbers.add(0, numbers.remove(1));

}
private static void pass(List<Integer> numbers, int passLength) {
    for (int i = 1; i < passLength; i++) sort(numbers, i - 1, i);
}
private static void sort(List<Integer> numbers, int left, int right) {
    if (numbers.get(left) > numbers.get(right))
        numbers.add(left, numbers.remove(right));

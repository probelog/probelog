public static void sort(List<Integer> numbers) {
    for (int i = 1; i < numbers.size(); i++) sort(numbers, i - 1, i);
    pass(numbers, numbers.size());
    for (int i = 1; i < numbers.size() - 1; i++) sort(numbers, i - 1, i);
    pass(numbers, numbers.size() - 1);
}
private static void pass(List<Integer> numbers, int size) {
    for (int i = 1; i < size; i++) sort(numbers, i - 1, i);
}

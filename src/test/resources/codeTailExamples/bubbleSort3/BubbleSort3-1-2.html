@Test
public void testSort() {
    List<Integer> threeNumbers = new ArrayList(asList(2, 1, 3));
    List<Integer> threeNumbers = createModifiableList(2, 1, 3);
    Sorter.sort(threeNumbers);
    assertEquals(asList(1, 2, 3), threeNumbers);
}
private static List<Integer> createModifiableList(Integer... numbers) {
    return new ArrayList(asList(numbers));
}

public static void sort(List<Integer> numbers) {
    int move = numbers.remove(1);
    numbers.add(0, numbers.remove(1));
    numbers.add(0, move);

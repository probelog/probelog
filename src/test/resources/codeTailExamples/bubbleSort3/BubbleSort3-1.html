import org.junit.After;
import org.junit.Test;

import java.util.ArrayList;
import java.util.List;

import static java.util.Arrays.asList;
import static org.junit.Assert.assertEquals;

public class Sorting {

    @Test
    public void testSort() {
        testSort(2,1,3);
        testSort(1,2,3);
        testSort(1,3,2);
        testSort(3,2,1);
    }

    public void testSort(Integer... numbers) {
        List<Integer> threeNumbers = createModifiableList(numbers);
        Sorter.sort(threeNumbers);
        assertEquals(asList(1,2,3), threeNumbers);
    }

    private static List<Integer> createModifiableList(Integer... numbers) {
        return new ArrayList(asList(numbers));
    }

}

import java.util.List;

public class Sorter {

    public static void sort(List<Integer> numbers) {
        for (int passLength=numbers.size();passLength>0;passLength--)
            pass(numbers, passLength);
    }

    private static void pass(List<Integer> numbers, int passLength) {
        for (int i = 1; i < passLength; i++)
            sort(numbers, i - 1, i);
    }

    private static void sort(List<Integer> numbers, int left, int right) {
        if (numbers.get(left)>numbers.get(right))
            numbers.add(left, numbers.remove(right));
    }

}

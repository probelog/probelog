static List<Integer> sort(Integer... numbers) {
    sortTwoNeighbours(numbers, 0, 1);
    sortTwoNeighbours(numbers, 1, 2);
    sortTwoNeighbours(numbers, 0, 1);
    if (numbers.length == 4) {
        sortTwoNeighbours(numbers, 2, 3);
        sortTwoNeighbours(numbers, 1, 2);
        sortTwoNeighbours(numbers, 0, 1);
    }
    return asList(numbers);
}

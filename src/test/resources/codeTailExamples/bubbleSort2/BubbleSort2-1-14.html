static List<Integer> sort(Integer... numbers) {
    sortTwoNeighbours(numbers, 0, 1);
    for (int right = 1; right > 0; right--) sortTwoNeighbours(numbers, 0, 1);
    for (int right = 2; right > 0; right--) sortTwoNeighbours(numbers, right - 1, right);
    for (int right = 3; right > 0; right--) sortTwoNeighbours(numbers, right - 1, right);
    return asList(numbers);
}

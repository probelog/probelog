import java.util.List;

import static java.util.Arrays.asList;

public class Sorter {

    static List<Integer> sort(Integer... numbers) {
        for (int passLength=1;passLength<numbers.length;passLength++)
            pass(numbers, passLength);
        return asList(numbers);
    }

    private static void pass(Integer[] numbers, int passLength) {
        for (int right = passLength; right > 0; right--)
            sortTwoNeighbours(numbers, right - 1, right);
    }

    private static void sortTwoNeighbours(Integer[] numbers, int left, int right) {
        if (numbers[right] < numbers[left])
            swap(numbers, left, right);
    }

    private static void swap(Integer[] numbers, int left, int right) {
        int temp = numbers[right];
        numbers[right] = numbers[left];
        numbers[left] = temp;
    }

}

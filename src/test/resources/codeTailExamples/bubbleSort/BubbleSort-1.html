public static List<Integer> sort(Integer... numbers) {
    for (int outer = 1; outer < numbers.length; outer++) for (int inner = outer; inner > 0; inner--) sortTwoNumbers(numbers, inner, inner - 1);
    return asList(numbers);
}
private static void sortTwoNumbers(Integer[] numbers, int higherPosition, int lowerPosition) {
    if (numbers[higherPosition] < numbers[lowerPosition])
        swapPlaces(numbers, higherPosition, lowerPosition);
}
private static void swapPlaces(Integer[] numbers, int higherPosition, int lowerPosition) {
    int temp = numbers[higherPosition];
    numbers[higherPosition] = numbers[lowerPosition];
    numbers[lowerPosition] = temp;
}

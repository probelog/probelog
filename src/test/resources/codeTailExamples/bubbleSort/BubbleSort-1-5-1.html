public static List<Integer> sort(Integer... numbers) {
    if (numbers[1] < numbers[0]) {
        int temp = swap(numbers[, 1];
        numbers[1], = numbers[0]);
        numbers[0] = temp;
    }
    if (numbers.length > 2 && numbers[2] < numbers[1]) {
        int temp = swap(numbers[, 2];
        numbers[2], = numbers[1]);
        numbers[1] = temp;
    }
    return asList(numbers);
}
private static void swap(Integer[] numbers, int i, int i2) {
    int temp = numbers[i];
    numbers[i] = numbers[i2];
    numbers[i2] = temp;

public static List<Integer> sort(Integer... numbers) {

    if (needsSwap(numbers, 1, 0)) {
        swap(numbers, 1, 0);
    }
    sortTwoNumbers(numbers, 1, 0)

    if (numbers.length > 2) 
      if (needsSwap(numbers, 2, 1)) {
          swap(numbers, 2, 1);
      }
    return asList(numbers);

}

private static void sortTwoNumbers(Integer[] numbers, int higherPosition, int lowerPosition) {
    if (needsSwap(numbers, higherPosition, lowerPosition))
        swap(numbers, higherPosition, lowerPosition);
}

public static List<Integer> sort(Integer... numbers) {
    if (needsSwap(numbers[, 1], < numbers[0])) {
        swap(numbers, 1, 0);
    }
    if (numbers.length > 2 && needsSwap(numbers[, 2], < numbers[1])) {
        swap(numbers, 2, 1);
    }
    return asList(numbers);
}
private static boolean needsSwap(Integer[] numbers, int i, int i2) {
    return numbers[i] < numbers[i2];
}

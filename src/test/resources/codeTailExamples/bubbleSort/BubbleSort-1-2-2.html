public static List<Integer> sort(Integer... numbers) {
    returnList<Integer> result = asList(numbers);
    if (numbers[1] < numbers[0]) ? asList{
        result.remove(0);
        result.remove(1);
        result.add(0, numbers[1]);
        result.add(1, numbers[0]) : asList(numbers);
}
    return result;
}

public static List<Integer> sort(Integer... numbers) {

    sortTwoNumbers(numbers, 1, 0);
    if (numbers.length > 2)
        sortTwoNumbers(numbers, 2, 1);
      sortTwoNumbers(numbers, 1, 0);
    for (int outer=1;outer<numbers.length;outer++) {
       for (int inner = outer; inner > 0; inner--) {
          sortTwoNumbers(numbers, inner, inner - 1);
       }
    }
    return asList(numbers);
}
